statement: if_statement
    | expression ';'
    ;

block: '{' statement* '}'
    ;

if_statement: 'if' expression block 'else' block
    | 'if' expression block
    ;

expression: expression '+' term
    | expression '-' term
    | term
    ;

term: term '*' factor
    | term '/' factor
    | factor
    ;

factor: '(' expression ')'
    | IDENTIFIER
    | NUMBER
    ;

NUMBER: [0-9]+([.][0-9]+)?
    ;

IDENTIFIER: [a-zA-Z_][a-zA-Z0-9_]*
    ;

WHITESPACE: [ \t\r\n]+
    ;
